68
69 // Calculate the start and end points
70 switch (lineCollider.direction)
71 {
72 case 0: // X-axis
73 startPoint = collision.transform.position - (collision.transform.right * lineLength / 2);

75 break;
76 case 1: // Y-axis
77 startPoint = collision.transform.position - (collision.transform.up * lineLength / 2);
78 endPoint = collision.transform.position + (collision.transform.up * lineLength / 2);
79 break;
80 case 2: // Z-axis
