75 private void calculateLineDirection(Collision collision)
76 {
77 CapsuleCollider lineCollider = collision.gameObject.GetComponent<CapsuleCollider>();
78 if (lineCollider == null) return; // Early exit if the collider is not found
79
80 // Calculate the line's direction based on its orientation
81 Vector3 lineVector = Vector3.zero;
82 switch (lineCollider.direction)
83 {
84 case 0: // X-axis
85 lineVector = collision.transform.right;
86 break;
87 case 1: // Y-axis
88 lineVector = collision.transform.up;
89 break;
90 case 2: // Z-axis
91 lineVector = collision.transform.forward;
92 break;
93 }
94
95 // Calculate the start and end points of the line
96 Vector3 startPoint = collision.transform.position - (lineVector * lineCollider.height / 2);
97 Vector3 endPoint = collision.transform.position + (lineVector * lineCollider.height / 2);
98
99 // Adjust for the collider's center offset
100 startPoint += lineCollider.center;
101 endPoint += lineCollider.center;
102
103 // Calculate the direction of the line
104 lineDirection = (endPoint - startPoint).normalized;
105 }
